$(".desktop-container").append(`<div class="context-menu-wrapper d-none"><div class="refresh-context"><span id="refresh">Обновить</span></div><div class="create-context"><span id="create">Создать</span></div></div>`)
let rectWrapper = document.querySelector(".rect-wrapper");
let startupAnimation = document.querySelector(".startup-anim");
let desktop = document.querySelector(".desktop-container");
let taskbar = document.querySelector(".taskbar-container");

function start() {
    function startMoving() {
        rectWrapper.style.transform = "translateX(-960px)"
    }

    function showAnim() {
        startupAnimation.classList.remove("d-none");
        setTimeout(startMoving, 1);
    }

    function hideAnim() {
        startupAnimation.classList.add("d-none")
    }

    function changeBGColor() {
        desktop.style.backgroundColor = "#008080";
    }

    function showDesktop() {
        desktop.classList.remove("d-none");
        taskbar.classList.remove("d-none");
    }
    setTimeout(showAnim, 500)
    setTimeout(hideAnim, 3150)
    setTimeout(changeBGColor, 6000)
    setTimeout(showDesktop, 6800)
}

window.onload = () => {
    start()
}

//TIME
let time = document.querySelector("#time")

function renderTime() {
    let date = new Date();
    let hours = date.getHours().toString().length == 1 ? `0${date.getHours()}` : date.getHours();
    let mins = date.getMinutes().toString().length == 1 ? `0${date.getMinutes()}` : date.getMinutes();
    time.innerHTML = `${hours}:${mins}`;
}

renderTime();
setInterval(renderTime, 500);

let menu = document.querySelector(".context-menu-wrapper");
let refreshBtn = document.querySelector(".refresh-context");
document.oncontextmenu = (e) => {
	if(desktop.classList.contains("d-none")) return false;
	if(e.clientX + 100 > window.innerWidth){
		menu.style.transform = `translate(${window.innerWidth - 100}px, ${e.clientY - 131}px)`
		menu.classList.remove("d-none");
		return false;
	}
	menu.style.transform = `translate(${e.clientX}px, ${e.clientY - 131}px)`
	menu.classList.remove("d-none");
    return false;
}
refreshBtn.addEventListener("click", () => {
	menu.classList.add("d-none");
})

document.addEventListener("click", () => {
	if(!menu.classList.contains("d-none")) menu.classList.add("d-none");
})

    }

        $("#frame").remove();
        ieWindow.removeClass("has-iframe");
    }
})
